released:2011, 
tagline:'When patriots become heroes'})

CREATE (Jones:Person {name:'Tommy Lee Jones', born:1946})
CREATE (Jackson:Person {name:'Samuel L. Jackson', born:1948})
CREATE (Atwell:Person {name:'Hayley Atwell', born:1982})
CREATE (Stan:Person {name:'Sebastian Stan', born:1982})
CREATE (Johnston:Person {name:'Joe Johnston', born:1950})
CREATE (Markus:Person {name:'Christopher Markus', born:1970})
CREATE (McFeely:Person {name:'Stephen McFeely', born:1970})

    (Evans)-[:ACTED_IN {roles:['Capitan America']}]->(CaptainAmericaFA),
    (Jackson)-[:ACTED_IN {roles:['Nick Fury']}]->(CaptainAmericaFA),
    (Atwell)-[:ACTED_IN {roles:['Peggy Carter']}]->(CaptainAmericaFA),
    (Stan)-[:ACTED_IN {roles:['James Buchanan']}]->(CaptainAmericaFA),
    (Jones)-[:ACTED_IN {roles:['Chester Philips']}]->(CaptainAmericaFA),
    (Johnston)-[:DIRECTED]->(CaptainAmericaFA),
    (Markus)-[:WROTE]->(CaptainAmericaFA),
    (McFeely)-[:WROTE]->(CaptainAmericaFA)
